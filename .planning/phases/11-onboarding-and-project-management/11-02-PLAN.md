config_store.py implementation:
```python
import json
from pathlib import Path
import platformdirs

APP_NAME = "AlteryxGitCompanion"

def _config_path() -> Path:
    data_dir = Path(platformdirs.user_data_dir(APP_NAME))
    data_dir.mkdir(parents=True, exist_ok=True)
    return data_dir / "config.json"

def load_config() -> dict:
    p = _config_path()
    if not p.exists():
        return {"version": 1, "projects": [], "active_project": None}
    return json.loads(p.read_text(encoding="utf-8"))

def save_config(cfg: dict) -> None:
    _config_path().write_text(
        json.dumps(cfg, indent=2, ensure_ascii=False), encoding="utf-8"
    )
```

Implement `app/services/git_ops.py`:
```python
import subprocess
from pathlib import Path

def is_git_repo(folder: str) -> bool:
    result = subprocess.run(
        ["git", "-C", folder, "rev-parse", "--git-dir"],
        capture_output=True, text=True
    )
    return result.returncode == 0

def git_init(folder: str) -> None:
    subprocess.run(
        ["git", "-C", folder, "init"],
        capture_output=True, text=True, check=True
    )

def get_git_identity() -> dict:
    def _get(key: str) -> str | None:
        r = subprocess.run(
            ["git", "config", "--global", key],
            capture_output=True, text=True
        )  # check=False intentional — exit 1 means key not set, not an error
        return r.stdout.strip() or None
    return {"name": _get("user.name"), "email": _get("user.email")}

def set_git_identity(name: str, email: str) -> None:
    subprocess.run(["git", "config", "--global", "user.name", name], check=True)
    subprocess.run(["git", "config", "--global", "user.email", email], check=True)
```

Implement `app/routers/projects.py` — replace 501 stubs with real logic. IMPORTANT: define `check_project` route BEFORE `remove_project` (the `/{project_id}` catch-all) to avoid routing ambiguity:

```python
from fastapi import APIRouter, HTTPException
from pydantic import BaseModel
from pathlib import Path
import uuid
from app.services.config_store import load_config, save_config
from app.services.git_ops import is_git_repo, git_init

router = APIRouter(prefix="/api/projects", tags=["projects"])

class ProjectIn(BaseModel):
    path: str

@router.get("")
def list_projects():
    cfg = load_config()
    return cfg.get("projects", [])

@router.get("/check")
def check_project(path: str):
    """Pre-flight check used by frontend before calling POST /api/projects.
    Returns {is_git_repo: bool} so the frontend can show a confirmation dialog
    before running git init."""
    return {"is_git_repo": is_git_repo(path)}

@router.post("", status_code=201)
def add_project(body: ProjectIn):
    p = Path(body.path)
    if not p.exists():
        raise HTTPException(status_code=400, detail="Path does not exist")
    resolved = str(p.resolve())
    cfg = load_config()
    if any(proj["path"] == resolved for proj in cfg.get("projects", [])):
        raise HTTPException(status_code=409, detail="Project already registered")
    if not is_git_repo(resolved):
        git_init(resolved)
    project = {"id": str(uuid.uuid4()), "path": resolved, "name": p.name}
    cfg.setdefault("projects", []).append(project)
    save_config(cfg)
    return project

@router.delete("/{project_id}")
def remove_project(project_id: str):
    cfg = load_config()
    projects = cfg.get("projects", [])
    match = next((p for p in projects if p["id"] == project_id), None)
    if not match:
        raise HTTPException(status_code=404, detail="Project not found")
    cfg["projects"] = [p for p in projects if p["id"] != project_id]
    if cfg.get("active_project") == project_id:
        cfg["active_project"] = None
    save_config(cfg)
    return {"removed": project_id}
```

Implement `app/routers/folder_picker.py` — tkinter import INSIDE function:
```python
import asyncio
from fastapi import APIRouter

router = APIRouter(prefix="/api/folder-picker", tags=["folder-picker"])

def _pick_folder() -> str | None:
    """Run in a thread (blocking). tkinter imported here to prevent headless CI crash."""
    try:
        import tkinter as tk
        from tkinter import filedialog
        root = tk.Tk()
        root.withdraw()
        root.wm_attributes("-topmost", True)  # Required on Windows: brings dialog in front of browser
        try:
            path = filedialog.askdirectory(title="Select Workflows Folder")
        finally:
            root.destroy()
        return path or None
    except Exception:
        return None

@router.post("")
async def pick_folder():
    selected = await asyncio.to_thread(_pick_folder)
    if selected is None:
        return {"path": None, "cancelled": True}
    return {"path": selected, "cancelled": False}
```
