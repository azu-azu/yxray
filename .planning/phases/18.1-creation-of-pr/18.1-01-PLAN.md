    ```python
    import re  # add if not present

    def parse_github_owner_repo(repo_url: str) -> tuple[str, str]:
        m = re.match(r"https://github\.com/([^/]+)/([^/]+?)(?:\.git)?$", repo_url)
        if not m:
            raise ValueError(f"Cannot parse GitHub URL: {repo_url}")
        return m.group(1), m.group(2)

    def create_pull_request(
        token: str, owner: str, repo: str, title: str, head: str,
        base: str = "main", body: str = "",
    ) -> dict:
        resp = httpx.post(
            f"{GITHUB_API_BASE}/repos/{owner}/{repo}/pulls",
            headers=_github_headers(token),
            json={"title": title, "head": head, "base": base, "body": body},
        )
        resp.raise_for_status()
        return resp.json()

    def get_open_pr_for_branch(token: str, owner: str, repo: str, branch: str) -> dict | None:
        resp = httpx.get(
            f"{GITHUB_API_BASE}/repos/{owner}/{repo}/pulls",
            headers=_github_headers(token),
            params={"head": f"{owner}:{branch}", "state": "open"},
        )
        resp.raise_for_status()
        items = resp.json()
        return items[0] if items else None
    ```

    ```python
    import re  # add if not present

    def parse_gitlab_namespace_path(repo_url: str) -> str:
        m = re.match(r"https://gitlab\.com/(.+?)(?:\.git)?$", repo_url)
        if not m:
            raise ValueError(f"Cannot parse GitLab URL: {repo_url}")
        return m.group(1)

    def get_gitlab_project_id(token: str, namespace_path: str) -> int:
        encoded = namespace_path.replace("/", "%2F")
        resp = httpx.get(
            f"{GITLAB_BASE}/projects/{encoded}",
            headers={"PRIVATE-TOKEN": token},
        )
        resp.raise_for_status()
        return resp.json()["id"]

    def create_merge_request(
        token: str, project_id: int, title: str, source_branch: str,
        target_branch: str = "main", description: str = "",
    ) -> dict:
        resp = httpx.post(
            f"{GITLAB_BASE}/projects/{project_id}/merge_requests",
            headers={"PRIVATE-TOKEN": token},
            json={
                "title": title,
                "source_branch": source_branch,
                "target_branch": target_branch,
                "description": description,
            },
        )
        resp.raise_for_status()
        return resp.json()

    def get_open_mr_for_branch(token: str, project_id: int, branch: str) -> dict | None:
        resp = httpx.get(
            f"{GITLAB_BASE}/projects/{project_id}/merge_requests",
            headers={"PRIVATE-TOKEN": token},
            params={"source_branch": branch, "state": "opened"},
        )
        resp.raise_for_status()
        items = resp.json()
        return items[0] if items else None
    ```
