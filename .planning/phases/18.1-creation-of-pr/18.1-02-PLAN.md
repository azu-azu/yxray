    ```python
    class PRCreateRequest(BaseModel):
        project_id: str
        folder: str
        provider: str = "github"
        title: str
        description: str = ""
        branch: str

    @router.post("/pr/create")
    def pr_create(body: PRCreateRequest) -> dict:
        try:
            if body.provider == "github":
                token = remote_auth.get_token(SERVICE_GITHUB)
                repo_url = config_store.get_remote_repo(body.project_id, "github")
                owner, repo = github_api.parse_github_owner_repo(repo_url)
                result = github_api.create_pull_request(
                    token, owner, repo, body.title, body.branch, body="", body_=body.description
                )
                # Note: 'body' param name conflicts with FastAPI body — use keyword arg
                pr_url = result["html_url"]
            else:  # gitlab
                token = remote_auth.get_token(SERVICE_GITLAB)
                repo_url = config_store.get_remote_repo(body.project_id, "gitlab")
                namespace = gitlab_api.parse_gitlab_namespace_path(repo_url)
                project_id = gitlab_api.get_gitlab_project_id(token, namespace)
                result = gitlab_api.create_merge_request(
                    token, project_id, body.title, body.branch, description=body.description
                )
                pr_url = result["web_url"]
            return {"pr_url": pr_url}
        except Exception as e:
            return {"success": False, "error": str(e)}

    @router.get("/pr/status")
    def pr_status(
        folder: str,
        project_id: str = "",
        provider: str = "github",
        branch: str = "",
    ) -> dict:
        try:
            if provider == "github":
                token = remote_auth.get_token(SERVICE_GITHUB)
                repo_url = config_store.get_remote_repo(project_id, "github")
                if not token or not repo_url or not branch:
                    return {"pr_exists": False, "pr_url": None}
                owner, repo = github_api.parse_github_owner_repo(repo_url)
                pr = github_api.get_open_pr_for_branch(token, owner, repo, branch)
                return {"pr_exists": pr is not None, "pr_url": pr["html_url"] if pr else None}
            else:
                token = remote_auth.get_token(SERVICE_GITLAB)
                repo_url = config_store.get_remote_repo(project_id, "gitlab")
                if not token or not repo_url or not branch:
                    return {"pr_exists": False, "pr_url": None}
                namespace = gitlab_api.parse_gitlab_namespace_path(repo_url)
                project_id_num = gitlab_api.get_gitlab_project_id(token, namespace)
                mr = gitlab_api.get_open_mr_for_branch(token, project_id_num, branch)
                return {"pr_exists": mr is not None, "pr_url": mr["web_url"] if mr else None}
        except Exception:
            return {"pr_exists": False, "pr_url": None}
    ```
