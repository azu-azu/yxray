```python
import json
import urllib.request

MARKER = "<!-- acd-diff-report -->"
_API_BASE = os.environ.get("CI_API_V4_URL", "https://gitlab.com/api/v4")
_PROJECT_ID = os.environ.get("CI_PROJECT_ID", "")
_MR_IID = os.environ.get("CI_MERGE_REQUEST_IID", "")
_TOKEN = os.environ.get("GITLAB_TOKEN", "")


def _note_headers() -> dict:
    return {"PRIVATE-TOKEN": _TOKEN, "Content-Type": "application/json"}


def list_notes() -> list:
    url = f"{_API_BASE}/projects/{_PROJECT_ID}/merge_requests/{_MR_IID}/notes?per_page=100"
    req = urllib.request.Request(url, headers=_note_headers())
    with urllib.request.urlopen(req, timeout=10) as resp:
        return json.loads(resp.read())


def post_note(body: str) -> None:
    url = f"{_API_BASE}/projects/{_PROJECT_ID}/merge_requests/{_MR_IID}/notes"
    data = json.dumps({"body": body}).encode()
    req = urllib.request.Request(url, data=data, headers=_note_headers(), method="POST")
    urllib.request.urlopen(req, timeout=10)


def update_note(note_id: int, body: str) -> None:
    url = f"{_API_BASE}/projects/{_PROJECT_ID}/merge_requests/{_MR_IID}/notes/{note_id}"
    data = json.dumps({"body": body}).encode()
    req = urllib.request.Request(url, data=data, headers=_note_headers(), method="PUT")
    urllib.request.urlopen(req, timeout=10)


def post_or_update_note(body: str) -> None:
    token = os.environ.get("GITLAB_TOKEN", "")
    if not token:
        print("GITLAB_TOKEN not set — skipping comment post")
        return
    notes = list_notes()
    existing = next((n for n in notes if MARKER in n.get("body", "")), None)
    if existing:
        update_note(existing["id"], body)
        print(f"Updated existing MR note {existing['id']}")
    else:
        post_note(body)
        print("Created new MR note")
```
